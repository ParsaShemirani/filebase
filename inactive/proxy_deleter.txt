import shutil

from sqlalchemy import select

from models import File, FileRelationship
from connection import Session
from env_vars import STORAGE_PATH

with Session() as session:
	with session.begin():
		all_proxy_files = session.scalars(
			select(File).join(FileRelationship, File.id == FileRelationship.source_id).where(FileRelationship.kind == "proxy_of")
		).all()


		all_proxy_relationships = session.scalars(select(FileRelationship).join(File, FileRelationship.source_id == File.id).where(FileRelationship.kind == "proxy_of")).all()

		for f in all_proxy_files:
			print(f"Proxy file ID: {f.id}")

			proxy_file_path = STORAGE_PATH / f"{f.id}.{f.extension}"

			print(f"File Path: {proxy_file_path}")

			session.delete(f)
			proxy_file_path.unlink()

			print("Deleted!")


		for r in all_proxy_relationships:
			print(f"Deleting: {r}")
			session.delete(r)
